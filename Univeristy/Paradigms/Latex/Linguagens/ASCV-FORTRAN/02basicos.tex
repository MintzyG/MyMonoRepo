    FUNCTION Area(x,y,z)
     IMPLICIT NONE
     REAL :: Area            ! tipo de funcao
     REAL, INTENT( IN ) :: x, y, z
     REAL :: theta, height
     theta = ACOS((x**2+y**2-z**2)/(2.0*x*y))
     height = x*SIN(theta); Area = 0.5*y*height
    END FUNCTION Area
